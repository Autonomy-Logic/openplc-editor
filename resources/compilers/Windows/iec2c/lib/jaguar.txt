FUNCTION_BLOCK ADC_CONFIG
VAR_INPUT
ADC_CH : SINT;
ADC_TYPE : SINT;
END_VAR
VAR_OUTPUT
SUCCESS : BOOL;
END_VAR
VAR
ADC_CH_LOCAL : SINT;
ADC_TYPE_LOCAL : SINT;
END_VAR

  IF ADC_CH <> ADC_CH_LOCAL OR ADC_TYPE <> ADC_TYPE_LOCAL THEN
    ADC_CH_LOCAL := ADC_CH;
    ADC_TYPE_LOCAL := ADC_TYPE;
    SUCCESS := TRUE;
  ELSE
    SUCCESS := FALSE;
  END_IF;

END_FUNCTION_BLOCK
